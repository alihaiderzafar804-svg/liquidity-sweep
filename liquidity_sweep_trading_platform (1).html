Strategy Logic — Liquidity Sweep SELL
Strict 3-candle pattern conditions
// Candle 1 — Bearish Setup
C1_bearish: close[2] < open[2]

// Candle 2 — Bullish Sweep
C2_bullish: close[1] > open[1]
C2_sweep: low[1] < low[2] // grabs liquidity
C2_wick: high[1] <= high[2] // wick filter
C2_close: close[1] > close[2] AND
close[1] < open[2] // inside range

// Candle 3 — Breakdown Entry
C3_break: close[0] < close[2] // confirm sell

SELL = ALL conditions TRUE

// Risk Levels
Entry = close[0]
SL = high[1] + buffer
TP = Entry - (SL - Entry) × RR
